function secondsOfDay() {
    let today = new Date();
    let hoursInSeconds = today.getHours() * 3600;
    let minutesInSeconds = today.getMinutes() * 60;
    let seconds = today.getSeconds();
    let time = hoursInSeconds + minutesInSeconds + seconds;
    return time;
}
